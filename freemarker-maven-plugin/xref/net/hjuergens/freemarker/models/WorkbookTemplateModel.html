3   import freemarker.template.*;
4   import org.apache.poi.ss.usermodel.Workbook;
5   
6   /**
7    * created on 29.09.15
8    *
9    * @author juergens

11  class WorkbookTemplateModel
12          extends WrappingTemplateModel
13          implements TemplateSequenceModel, TemplateHashModel, AdapterTemplateModel  {
14  
15      private final Workbook workbook;
16  
17      public WorkbookTemplateModel(Workbook workbook, ObjectWrapper ow) {
18          super(ow);  // coming from WrappingTemplateModel
19          this.workbook = workbook;
20      }
21  
22      @Override  // coming from TemplateSequenceModel
23      public int size() throws TemplateModelException {
24          return workbook.getNumberOfSheets();
25      }
26  
27      @Override  // coming from TemplateSequenceModel
28      public TemplateModel get(int index) throws TemplateModelException {
29          return wrap( workbook.getSheetAt(index) );
30      }
31  
32      @Override  // coming from AdapterTemplateModel
33      public Object getAdaptedObject(Class hint) {
34          return workbook;
35      }
36  
37      @Override
38      public TemplateModel get(String s) throws TemplateModelException {
39          if(s.equals("workbook"))
40              return wrap(workbook);
41          else
42              return wrap( workbook.getSheet(s) );
43      }
44  
45      @Override
46      public boolean isEmpty() throws TemplateModelException {
47          return size() > 0;
48      }
49  }
